<script>
(function(){
    // DOM elements
    const sections = document.querySelectorAll('.section');
    const prevBtn = document.getElementById('prevBtn');
    const nextBtn = document.getElementById('nextBtn');
    const menuToggle = document.getElementById('menuToggleBtn');
    const overlayMenu = document.getElementById('overlayMenu');
    const closeMenuBtn = document.getElementById('closeMenuBtn');
    const menuLinks = document.querySelectorAll('.menu-links button');
    const transitionOverlay = document.getElementById('transitionOverlay');
    
    let currentIndex = 0;
    let isTransitioning = false;
    let pendingIndex = null;
    const totalSections = sections.length;

    // Helper: reveal staged elements with cascade
    function revealStagedInSection(sectionElement) {
      if (!sectionElement) return;
      const stagedEls = sectionElement.querySelectorAll('.staged');
      stagedEls.forEach(el => el.classList.remove('revealed'));
      stagedEls.forEach((el, idx) => {
        setTimeout(() => {
          if (sectionElement.classList.contains('active')) {
            el.classList.add('revealed');
          }
        }, idx * 100);
      });
    }

    // Immediately switch active section (no animation, just DOM update)
    function setActiveSectionImmediately(index) {
      sections.forEach((sec, i) => {
        sec.classList.remove('active');
      });
      const targetSection = sections[index];
      targetSection.classList.add('active');
      currentIndex = index;
      // Reset scroll position
      targetSection.scrollTop = 0;
      // Reveal staged content
      revealStagedInSection(targetSection);
    }

    // Core transition with black panel - FIXED: no flash of old content
    function navigateToSection(newIndex) {
      if (isTransitioning) return;
      if (newIndex === currentIndex) return;
      if (newIndex < 0 || newIndex >= totalSections) return;
      
      isTransitioning = true;
      document.body.classList.add('transition-active');
      
      // CRITICAL FIX: Immediately hide the old section and show new section
      // but keep them both with proper z-index handling
      // First, prepare target section (make it visible but behind overlay)
      const targetSection = sections[newIndex];
      const currentSection = sections[currentIndex];
      
      // Reset staged on target section
      targetSection.querySelectorAll('.staged').forEach(el => el.classList.remove('revealed'));
      
      // Make target section active but with opacity 0? No - better: 
      // We'll make it active and fully visible, but the black overlay covers everything.
      // The problem was old section flashing AFTER animation. Solution: 
      // Switch active class BEFORE animation starts, so new section is ready.
      // But we must ensure the new section doesn't appear abruptly.
      // Since black overlay covers the screen, user won't see the switch.
      
      // Switch active class IMMEDIATELY (while black overlay will cover)
      sections.forEach(sec => sec.classList.remove('active'));
      targetSection.classList.add('active');
      
      // Reset scroll for target
      targetSection.scrollTop = 0;
      
      // Store old index for reference
      const oldIndex = currentIndex;
      currentIndex = newIndex;
      
      // Now trigger black panel animation
      transitionOverlay.classList.remove('animate-sweep');
      // Force reflow
      void transitionOverlay.offsetWidth;
      transitionOverlay.classList.add('animate-sweep');
      
      // After animation completes, finalize and reveal staged content
      const onAnimationEnd = () => {
        transitionOverlay.classList.remove('animate-sweep');
        
        // Reveal staged elements with beautiful cascade
        revealStagedInSection(targetSection);
        
        // Cleanup
        isTransitioning = false;
        document.body.classList.remove('transition-active');
        transitionOverlay.removeEventListener('animationend', onAnimationEnd);
      };
      
      transitionOverlay.addEventListener('animationend', onAnimationEnd, { once: true });
      
      // Fallback timeout (in case animationend doesn't fire)
      setTimeout(() => {
        if (isTransitioning) {
          transitionOverlay.classList.remove('animate-sweep');
          revealStagedInSection(targetSection);
          isTransitioning = false;
          document.body.classList.remove('transition-active');
        }
      }, 1000);
    }
    
    // Next / Prev functions
    function nextSection() {
      if (currentIndex + 1 < totalSections) {
        navigateToSection(currentIndex + 1);
      }
    }
    
    function prevSection() {
      if (currentIndex - 1 >= 0) {
        navigateToSection(currentIndex - 1);
      }
    }
    
    // Event listeners for bottom navigation
    prevBtn.addEventListener('click', prevSection);
    nextBtn.addEventListener('click', nextSection);
    
    // Keyboard navigation
    window.addEventListener('keydown', (e) => {
      if (e.target.tagName === 'INPUT' || e.target.tagName === 'TEXTAREA') return;
      if (e.key === 'ArrowLeft') {
        e.preventDefault();
        prevSection();
      } else if (e.key === 'ArrowRight') {
        e.preventDefault();
        nextSection();
      } else if (e.key === 'Escape' && overlayMenu.classList.contains('active')) {
        overlayMenu.classList.remove('active');
      }
    });
    
    // Global menu overlay
    function openMenu() { overlayMenu.classList.add('active'); }
    function closeMenu() { overlayMenu.classList.remove('active'); }
    menuToggle.addEventListener('click', openMenu);
    closeMenuBtn.addEventListener('click', closeMenu);
    overlayMenu.addEventListener('click', (e) => { if (e.target === overlayMenu) closeMenu(); });
    
    // Menu navigation links
    menuLinks.forEach(link => {
      link.addEventListener('click', (e) => {
        const targetIdx = parseInt(link.getAttribute('data-nav'), 10);
        if (!isNaN(targetIdx) && targetIdx !== currentIndex) {
          navigateToSection(targetIdx);
        }
        closeMenu();
      });
    });
    
    // Hero begin journey button
    const beginBtn = document.querySelector('.next-trigger');
    if (beginBtn) {
      beginBtn.addEventListener('click', () => { 
        if (currentIndex === 0) nextSection(); 
      });
    }
    
    // Contact form demo
    const contactForm = document.getElementById('demo-contact-form');
    if (contactForm) {
      contactForm.addEventListener('submit', (e) => {
        e.preventDefault();
        const btn = contactForm.querySelector('.btn-submit');
        const originalText = btn.innerText;
        btn.innerText = 'Sent ✓';
        setTimeout(() => { btn.innerText = originalText; }, 1200);
        contactForm.reset();
      });
    }
    
    // Initial staged reveal for active section
    function initializeStaged() {
      const activeSection = document.querySelector('.section.active');
      if (activeSection) {
        const stagedItems = activeSection.querySelectorAll('.staged');
        stagedItems.forEach(el => el.classList.remove('revealed'));
        stagedItems.forEach((el, idx) => {
          setTimeout(() => {
            if (activeSection.classList.contains('active')) {
              el.classList.add('revealed');
            }
          }, idx * 100);
        });
      }
    }
    
    // Set initial state
    setActiveSectionImmediately(0);
    initializeStaged();
    
    // Prevent body scroll
    document.body.style.overflow = 'hidden';
    
    // Ensure no scroll-based navigation
    window.addEventListener('wheel', (e) => {
      // Do nothing - prevents accidental scroll navigation
    }, { passive: false });
    
    // Micro-interaction: button hover effects are in CSS
    // Additional smoothness for all interactive elements
    const allButtons = document.querySelectorAll('button');
    allButtons.forEach(btn => {
      btn.addEventListener('mousedown', () => {
        if (!btn.classList.contains('nav-btn') && !btn.classList.contains('menu-btn') && !btn.classList.contains('close-menu')) return;
        btn.style.transform = 'scale(0.96)';
        setTimeout(() => { btn.style.transform = ''; }, 120);
      });
    });
    
    // Optional: preload sections for smoother experience
    // All sections already in DOM, good to go
})();
</script>
