<script>
(function(){
    // ---------- DOM elements ----------
    const sections = document.querySelectorAll('.section');
    const prevBtn = document.getElementById('prevBtn');
    const nextBtn = document.getElementById('nextBtn');
    const menuToggle = document.getElementById('menuToggleBtn');
    const overlayMenu = document.getElementById('overlayMenu');
    const closeMenuBtn = document.getElementById('closeMenuBtn');
    const transitionOverlay = document.getElementById('transitionOverlay');
    const progressBar = document.getElementById('progressBar');
    const progressIndexSpan = document.getElementById('progressIndex');
    const fullscreenOverlay = document.getElementById('fullscreenOverlay');
    const closeOverlayBtn = document.getElementById('closeOverlayBtn');
    const overlayContentDiv = document.getElementById('overlayContent');
    
    let currentIndex = 0;
    let isTransitioning = false;
    const totalSections = sections.length;
    
    // ---------- LOCAL STORAGE MEMORY SYSTEM ----------
    function saveState() { localStorage.setItem('lastSectionIndex', currentIndex); }
    function loadStoredIndex() { const saved = localStorage.getItem('lastSectionIndex'); return saved !== null ? parseInt(saved, 10) : 0; }
    
    // ---------- PROGRESS UPDATE ----------
    function updateProgress() { progressBar.style.width = `${((currentIndex + 1) / totalSections) * 100}%`; progressIndexSpan.innerText = `${String(currentIndex+1).padStart(2,'0')} / ${String(totalSections).padStart(2,'0')}`; }
    
    // ---------- STAGED REVEAL (improved timing + line reveal) ----------
    function revealStagedInSection(sectionElement) {
      if (!sectionElement) return;
      const stagedEls = sectionElement.querySelectorAll('.staged');
      stagedEls.forEach(el => el.classList.remove('revealed'));
      // also handle line-reveal
      const lineHr = sectionElement.querySelector('hr.line-reveal');
      if(lineHr) lineHr.classList.remove('revealed');
      
      stagedEls.forEach((el, idx) => {
        setTimeout(() => {
          if (sectionElement.classList.contains('active')) {
            el.classList.add('revealed');
            // optional letter-by-letter effect on h1/h2
            if(el.querySelector('h1') && !el.querySelector('h1').hasAttribute('data-lettered')) {
              applyLetterEffect(el.querySelector('h1'));
            }
            if(el.querySelector('h2') && !el.querySelector('h2').hasAttribute('data-lettered')) {
              applyLetterEffect(el.querySelector('h2'));
            }
          }
        }, idx * 90);
      });
      setTimeout(() => { if(lineHr && sectionElement.classList.contains('active')) lineHr.classList.add('revealed'); }, stagedEls.length * 90 + 100);
    }
    
    function applyLetterEffect(heading) {
      if(heading.getAttribute('data-lettered')) return;
      const text = heading.innerText;
      heading.innerHTML = text.split('').map(char => char === ' ' ? ' ' : `<span style="animation-delay: ${Math.random() * 0.03}s">${char}</span>`).join('');
      heading.setAttribute('data-lettered', 'true');
      const spans = heading.querySelectorAll('span');
      spans.forEach((span, idx) => { span.style.animationDelay = `${idx * 0.02}s`; });
    }
    
    // ---------- SECTION NAVIGATION (with smooth overlay) ----------
    function setActiveSectionImmediately(index, skipStorage = false) {
      sections.forEach((sec, i) => { sec.classList.remove('active'); });
      const target = sections[index];
      target.classList.add('active');
      currentIndex = index;
      target.scrollTop = 0;
      revealStagedInSection(target);
      updateProgress();
      if(!skipStorage) saveState();
      applyDynamicBackground(index);
    }
    
    function navigateToSection(newIndex) {
      if(isTransitioning || newIndex === currentIndex || newIndex < 0 || newIndex >= totalSections) return;
      isTransitioning = true;
      document.body.classList.add('transition-active');
      const targetSection = sections[newIndex];
      sections.forEach(sec => sec.classList.remove('active'));
      targetSection.classList.add('active');
      targetSection.scrollTop = 0;
      const oldIndex = currentIndex;
      currentIndex = newIndex;
      updateProgress();
      saveState();
      applyDynamicBackground(newIndex);
      transitionOverlay.classList.remove('animate-sweep');
      void transitionOverlay.offsetWidth;
      transitionOverlay.classList.add('animate-sweep');
      const onEnd = () => {
        transitionOverlay.classList.remove('animate-sweep');
        revealStagedInSection(targetSection);
        isTransitioning = false;
        document.body.classList.remove('transition-active');
        transitionOverlay.removeEventListener('animationend', onEnd);
      };
      transitionOverlay.addEventListener('animationend', onEnd, { once: true });
      setTimeout(() => { if(isTransitioning) { transitionOverlay.classList.remove('animate-sweep'); revealStagedInSection(targetSection); isTransitioning=false; document.body.classList.remove('transition-active'); } }, 1000);
    }
    
    function nextSection() { if(currentIndex+1 < totalSections) navigateToSection(currentIndex+1); }
    function prevSection() { if(currentIndex-1 >= 0) navigateToSection(currentIndex-1); }
    
    // Dynamic background shift (already handled by CSS classes, but force update)
    function applyDynamicBackground(index) { document.documentElement.style.setProperty('--bg-primary', `var(--bg-section-${index})`); }
    
    // ---------- FULLSCREEN PROJECT DETAIL OVERLAY ----------
    function openProjectDetail(card) {
      const title = card.getAttribute('data-title') || card.querySelector('h4')?.innerText || 'Untitled';
      const year = card.getAttribute('data-year') || card.querySelector('.year')?.innerText || '—';
      const desc = card.getAttribute('data-desc') || 'No description available.';
      overlayContentDiv.innerHTML = `<h2>${title}</h2><div class="overlay-year">${year}</div><div class="overlay-desc">${desc}</div><p class="text-muted" style="margin-top:2rem;">A quiet exploration of space and material.</p>`;
      fullscreenOverlay.classList.add('active');
      document.body.style.overflow = 'hidden';
    }
    function closeProjectDetail() { fullscreenOverlay.classList.remove('active'); document.body.style.overflow = ''; }
    
    // ---------- PROJECT FILTER SYSTEM ----------
    function initFilter() {
      const filterBtns = document.querySelectorAll('.filter-btn');
      const projectsGrid = document.getElementById('projectsGrid');
      const allCards = Array.from(document.querySelectorAll('.project-card'));
      filterBtns.forEach(btn => {
        btn.addEventListener('click', () => {
          filterBtns.forEach(b => b.classList.remove('active'));
          btn.classList.add('active');
          const filterValue = btn.getAttribute('data-filter');
          allCards.forEach(card => {
            const year = card.getAttribute('data-year');
            const type = card.getAttribute('data-type');
            if(filterValue === 'all' || year === filterValue || type === filterValue) card.classList.remove('hidden-filter');
            else card.classList.add('hidden-filter');
          });
        });
      });
    }
    
    // ---------- TOUCH GESTURES (swipe) ----------
    let touchStartX = 0;
    document.addEventListener('touchstart', (e) => { touchStartX = e.changedTouches[0].screenX; }, { passive: true });
    document.addEventListener('touchend', (e) => {
      if(isTransitioning) return;
      const diff = e.changedTouches[0].screenX - touchStartX;
      if(Math.abs(diff) > 60) { diff > 0 ? prevSection() : nextSection(); }
    });
    
    // ---------- EASTER EGG (konami-style: up,up,down,down,left,right,left,right) ----------
    let easterEggSequence = [];
    const secretCode = ['ArrowUp','ArrowUp','ArrowDown','ArrowDown','ArrowLeft','ArrowRight','ArrowLeft','ArrowRight'];
    window.addEventListener('keydown', (e) => {
      easterEggSequence.push(e.key);
      if(easterEggSequence.length > secretCode.length) easterEggSequence.shift();
      if(JSON.stringify(easterEggSequence) === JSON.stringify(secretCode)) {
        alert('✨ Silence speaks volumes — thank you for the hidden journey ✨');
        easterEggSequence = [];
      }
    });
    
    // ---------- CUSTOM CURSOR ----------
    const cursor = document.getElementById('customCursor');
    if(cursor) {
      document.addEventListener('mousemove', (e) => { cursor.style.transform = `translate(${e.clientX}px, ${e.clientY}px)`; });
      const interactiveElements = document.querySelectorAll('button, .project-card, .nav-btn, .menu-btn, .filter-btn');
      interactiveElements.forEach(el => {
        el.addEventListener('mouseenter', () => { cursor.classList.add('hover-active'); });
        el.addEventListener('mouseleave', () => { cursor.classList.remove('hover-active', 'view-active'); });
      });
      const projectCards = document.querySelectorAll('.project-card');
      projectCards.forEach(card => {
        card.addEventListener('mouseenter', () => { cursor.classList.add('view-active'); });
        card.addEventListener('mouseleave', () => { cursor.classList.remove('view-active'); });
      });
    }
    
    // ---------- EVENT LISTENERS ----------
    prevBtn.addEventListener('click', prevSection);
    nextBtn.addEventListener('click', nextSection);
    window.addEventListener('keydown', (e) => {
      if(e.key === 'ArrowLeft') { e.preventDefault(); prevSection(); }
      if(e.key === 'ArrowRight') { e.preventDefault(); nextSection(); }
      if(e.key === 'Escape' && fullscreenOverlay.classList.contains('active')) closeProjectDetail();
      if(e.key === 'Escape' && overlayMenu.classList.contains('active')) overlayMenu.classList.remove('active');
    });
    
    // Menu overlay dynamic
    function buildMenu() { const container = document.getElementById('menuLinksContainer'); container.innerHTML = ''; sections.forEach((_, idx) => { const btn = document.createElement('button'); btn.innerText = sections[idx].querySelector('.chapter-label')?.innerText || `Section ${idx+1}`; btn.setAttribute('data-nav', idx); btn.addEventListener('click', () => { navigateToSection(idx); overlayMenu.classList.remove('active'); }); container.appendChild(btn); }); }
    menuToggle.addEventListener('click', () => overlayMenu.classList.add('active'));
    closeMenuBtn.addEventListener('click', () => overlayMenu.classList.remove('active'));
    overlayMenu.addEventListener('click', (e) => { if(e.target === overlayMenu) overlayMenu.classList.remove('active'); });
    buildMenu();
    
    document.querySelectorAll('.project-card').forEach(card => card.addEventListener('click', (e) => { e.stopPropagation(); openProjectDetail(card); }));
    closeOverlayBtn.addEventListener('click', closeProjectDetail);
    fullscreenOverlay.addEventListener('click', (e) => { if(e.target === fullscreenOverlay) closeProjectDetail(); });
    document.querySelector('.next-trigger')?.addEventListener('click', () => { if(currentIndex === 0) nextSection(); });
    document.getElementById('demo-contact-form')?.addEventListener('submit', (e) => { e.preventDefault(); alert('✨ Message received (demo)'); e.target.reset(); });
    
    // Initial load with local storage restore
    const savedIndex = loadStoredIndex();
    setActiveSectionImmediately(savedIndex, false);
    initFilter();
    updateProgress();
    applyDynamicBackground(currentIndex);
})();
